int	check_max(char c, int cmp, int pos, int *i, int max)
{
	if (c == '1')
	{
		if (cmp > pos && cmp - pos > max)
		{
			max = cmp - pos;
			*i = -1;
		}
		else if (pos - cmp > max)
		{
			max = pos - cmp;
			*i = 0;
		}
	}
	return (max);
}
